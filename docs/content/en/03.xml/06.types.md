## Element

The core type representing an XML element in the parsed tree:

```ts
interface Element {
  type: "element";
  name: string;
  attributes?: Attributes;
  elements?: Element[];
}
```

Text nodes use a different structure:

```ts
interface TextElement {
  type: "text";
  text: string;
}
```

### Creating Elements

```ts
const paragraph: Element = {
  type: "element",
  name: "w:p",
  elements: [
    {
      type: "element",
      name: "w:r",
      elements: [{ type: "text", text: "Hello World" }],
    },
  ],
};
```

## Attributes

```ts
interface Attributes {
  [key: string]: string | number | undefined;
}
```

A simple key-value map for XML attributes.

## XmlObject

The node-xml compatible format used by the `xml()` function:

```ts
interface XmlObject {
  [elementName: string]: (XmlObject | string)[];
}
```

Example:

```ts
const obj: XmlObject = {
  "w:p": [{ "w:r": [{ "w:t": ["Hello"] }] }],
};
```

## XmlAttrs

Attribute format for node-xml compatible objects:

```ts
interface XmlAttrs {
  [attrName: string]: string | number | boolean;
}
```

## Option Types

### Xml2JsOptions

Options for the `xml2js()` function:

```ts
interface Xml2JsOptions {
  trim?: boolean;
  sanitize?: boolean;
  nativeTypeAttributes?: boolean;
  ignoreDeclaration?: boolean;
  ignoreComment?: boolean;
  ignoreCdata?: boolean;
  ignoreDoctype?: boolean;
  ignoreText?: boolean;
  captureSpacesBetweenElements?: boolean;
  compact?: boolean;
}
```

### Js2XmlOptions

Options for the `js2xml()` function:

```ts
interface Js2XmlOptions {
  spaces?: number | string;
  fullTagEmptyElement?: boolean;
  ignoreDeclaration?: boolean;
  ignoreComment?: boolean;
  ignoreCdata?: boolean;
  ignoreDoctype?: boolean;
  ignoreText?: boolean;
  attributeValueFn?: (val: string, name: string) => string;
}
```
