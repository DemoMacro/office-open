---
title: "Animations"
description: "Add entrance, emphasis, and exit animations to shapes"
---

Add animation effects to shapes to control how they appear and behave during a presentation.

## Basic Animation

```ts
import { Slide, Shape, Animation, Paragraph, TextRun } from "@office-open/pptx";

new Slide({
    children: [
        new Shape({
            x: 1,
            y: 1,
            width: 6,
            height: 3,
            text: "Animated shape",
            animation: new Animation({
                type: "appear",
            }),
        }),
    ],
});
```

## Entrance Animations

### Appear

```ts
new Animation({ type: "appear" });
```

### Fly In

```ts
new Animation({
    type: "fly",
    dir: "fromLeft",
    // dir: "fromRight" | "fromTop" | "fromBottom"
});
```

### Fade In

```ts
new Animation({ type: "fadeIn" });
```

### Wipe

```ts
new Animation({
    type: "wipe",
    dir: "right",
    // dir: "left" | "up" | "down"
});
```

### Float In

```ts
new Animation({
    type: "float",
    dir: "up",
});
```

### Zoom

```ts
new Animation({
    type: "zoom",
    // preset: "in" | "out"
});
```

## Animation Options

```ts
new Animation({
    type: "fly",
    dir: "fromBottom",
    delay: 500, // Delay in milliseconds
    duration: 1000, // Duration in milliseconds
});
```

## Multiple Animated Shapes

Animate shapes sequentially by setting delays:

```ts
new Slide({
    children: [
        new Shape({
            x: 1,
            y: 1,
            width: 8,
            height: 1,
            text: "First item",
            animation: new Animation({ type: "fadeIn", delay: 0 }),
        }),
        new Shape({
            x: 1,
            y: 2.2,
            width: 8,
            height: 1,
            text: "Second item",
            animation: new Animation({ type: "fadeIn", delay: 500 }),
        }),
        new Shape({
            x: 1,
            y: 3.4,
            width: 8,
            height: 1,
            text: "Third item",
            animation: new Animation({ type: "fadeIn", delay: 1000 }),
        }),
    ],
});
```

## Full Example

```ts
import {
    Presentation,
    Slide,
    Shape,
    Paragraph,
    TextRun,
    Animation,
    Packer,
} from "@office-open/pptx";
import fs from "node:fs";

const pres = new Presentation({
    slides: [
        new Slide({
            children: [
                new Shape({
                    x: 1,
                    y: 1,
                    width: 8,
                    height: 2,
                    paragraphs: [
                        new Paragraph({
                            alignment: "center",
                            children: [
                                new TextRun({ text: "Animated Title", fontSize: 36, bold: true }),
                            ],
                        }),
                    ],
                    animation: new Animation({ type: "fly", dir: "fromBottom", duration: 800 }),
                }),
                new Shape({
                    x: 2,
                    y: 3.5,
                    width: 6,
                    height: 2,
                    paragraphs: [
                        new Paragraph({
                            alignment: "center",
                            children: [new TextRun({ text: "Content appears next", fontSize: 24 })],
                        }),
                    ],
                    animation: new Animation({ type: "fadeIn", delay: 800, duration: 600 }),
                }),
            ],
        }),
    ],
});

const buffer = await Packer.toBuffer(pres);
fs.writeFileSync("animated.pptx", buffer);
```
