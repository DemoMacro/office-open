---
title: "切换效果"
description: "添加淡入、推入、擦除等幻灯片切换效果"
---

使用幻灯片选项中的 `transition` 属性在切换幻灯片时添加动画效果。

## 基本切换

::api-example{type="pptx"}

```json [JSON]
{
  "children": [],
  "transition": { "type": "fade", "speed": "med" }
}
```

```ts [Class]
{
  children: [],
  transition: { type: "fade", speed: "med" },
}
```

::

## 切换类型

### 淡入

```ts
transition: {
  type: "fade";
}
```

### 推入

```ts
transition: { type: "push", dir: "r" }
// dir: "l"（左） | "r"（右） | "u"（上） | "d"（下）
```

### 擦除

```ts
transition: { type: "wipe", dir: "d" }
```

### 覆盖

```ts
transition: { type: "cover", dir: "r" }
```

### 拆分

```ts
transition: { type: "split", orient: "horz", dir: "out" }
// orient: "horz" | "vert"
// dir: "in" | "out"
```

### 轮辐

```ts
transition: { type: "wheel", spokes: 4 }
```

### 溶解

```ts
transition: { type: "dissolve", speed: "slow" }
```

### 随机

```ts
transition: {
  type: "random";
}
```

## 切换选项

| 属性               | 类型                        | 说明                       |
| ------------------ | --------------------------- | -------------------------- |
| `type`             | `TransitionType`            | 切换类型                   |
| `speed`            | `"slow" \| "med" \| "fast"` | 切换速度                   |
| `dir`              | `string`                    | 方向（取决于类型）         |
| `orient`           | `"horz" \| "vert"`          | 方向（拆分/百叶窗/棋盘等） |
| `spokes`           | `number`                    | 轮辐数量（轮辐切换）       |
| `advanceOnClick`   | `boolean`                   | 单击鼠标时切换             |
| `advanceAfterTime` | `number`                    | 自动切换间隔（毫秒）       |

### 可用类型

`"fade"` | `"push"` | `"wipe"` | `"split"` | `"blinds"` | `"checker"` | `"comb"` | `"randomBar"` | `"cover"` | `"pull"` | `"strips"` | `"wheel"` | `"zoom"` | `"circle"` | `"dissolve"` | `"diamond"` | `"newsflash"` | `"plus"` | `"wedge"` | `"random"` | `"cut"`

  "slides": [

            "height": 100,
            "text": "Welcome",
            "fill": "4472C4"

        }
      ],
      "transition": { "type": "fade", "speed": "med" }
    },
    {
      "children": [

          "shape": {

            "y": 150,

            "height": 100,

            "fill": "ED7D31"

        }

      "transition": { "type": "push", "dir": "r", "speed": "slow" }
    },
    {
      "children": [
